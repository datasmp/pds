Now, we will replace some values in the dataframe with missing values (NaN).

import random
import numpy as np

random.seed(0)

r = 0.4
a,b = df_stock.shape
missing = int(a*b*r)
for i in range( missing ):
    k = random.randint(2,b-1)
    p = random.randint(0,a-1)
    df_stock.iloc[p,k] = np.nan
